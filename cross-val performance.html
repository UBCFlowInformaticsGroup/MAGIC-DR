decision_rules <- function(df1, df2, df3, thresh1, thresh2, thresh3, tube3 = TRUE) {
  x1 = df1$Median_Blast_Prob
  x2 = df2$Median_Blast_Prob
  x3 = df3$Median_Blast_Prob
  rule1 = x1 >= thresh1
  rule1b = x2 >=thresh1
  rule2 = (x1 >= thresh2) & (x2 > thresh2)
  
  if (tube3) {
  rule1c = x3 >= thresh1
  rule2b = (x1 >= thresh2) & (x3 >= thresh2)
  rule2c = (x3 >= thresh2) & (x2 >= thresh2)
  rule3 = (x1 >= thresh3) & (x2 >= thresh3) & (x3 >= thresh3)
  combo_rule = rule1 | rule1b | rule1c | rule2 | rule2b | rule2c | rule3
  sensitivity = mean(combo_rule)
  false_negs = df1$Sample[!combo_rule]
  }
  else {
    sensitivity = mean(rule1 | rule1b | rule2)
    false_negs = df1$Sample[!(rule1 | rule1b | rule2)]
  }
  return(c(sensitivity, false_negs))
}
